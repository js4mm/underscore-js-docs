Collection Functions (Arrays or Objects)

each _.each(list , iterator, [context])forEach
list ထဲက element ေတြတစ္ခုျခင္းစီေပၚမွာ iterator ကအလုပ္လုပ္မွာပါforeach နဲ႔ အတူတူပါပဲ၊ iterator ရဲ႕ parameter ေတြကေတာ့Array ျဖစ္လွ်င္ ⇒(element, index, list) Object ျဖစ္လွ်င္ ⇒ (value, key, list) ျဖစ္ပါတယ္each method ကမူလ list ကိုပဲ return ျပန္ပါတယ္၊iterator က return ျပန္ထားတဲ့ data က အက်ိဳးသက္ေရာက္မႈ႔ရိွမွာမဟုတ္ပါဘူးဥပမာ -
_.each([1,2,3], function(el){alert(el);});// [or]_.each([1,2,3], alert);//=> alerts each number in turn..._.each({one:1, two:2, three:3},function(v, k){alert(k + ' => ' + v );});//=> alerts each key => value in turn...
မွတ္ခ်က္ : Collection function မ်ားသည္ arrays, objects နဲ႔Array-like objects(arguments,NodeList.. etc) ေတြေပၚမွာလုပ္လုပ္ပါတယ္ဒါေပမယ့္ duck-typing အေနနဲ႔ အလုပ္လုပ္တာေၾကာင့္numberic length property ပါေသာ Object မ်ား passing လုပ္ျခင္းကို လံုး၀ ေရွာင္ရွားရပါမည္။i.e: {name: 'koko', age: 20, length:3}
map _.map(list, iterator, [context])collect 
_.each နဲ႔ သေဘာတရားတူပါတယ္၊list ထဲက element တုိင္းေပၚမွာအလုပ္လုပ္မွာပါဒါေပမယ့္ _.each နွင့္မတူတာက iterator ထဲက return ျပန္ထားတဲ့ data ေတြအားလံုးကို array အျဖစ္နဲ႔ return ျပန္ေပးမွာျဖစ္ပါတယ္ ပထမ param က listing လုပ္မယ္ array or object ျဖစ္ပါတယ္ဒုတိယ param ကေတာ့ transformation function(iterator) ျဖစ္ၿပီး param အေနနဲ႔-Array ျဖစ္လွ်င္ ⇒ (element, index, list) Object ျဖစ္လွ်င္ ⇒ (value, key, list) အေနနဲ႔ passing လုပ္ပါတယ္javascript မွာ map ဆိုတဲ့ function ရိွၿပီးသားဆိုရင္အဲဒီnative map function ကိုပဲအသံုးျပဳသြားပါမယ္
_.map(['2', '22', '33'],function(v, k , l){return v;});//=> ['2', '22', '33'] _.map({one:1, two:2, three:3},function(v, k , l){return v;});//=> [1, 2, 3]
reduce _.reduce(list, iterator, memo, [context]) inject, foldl
Array ထဲက element ေတြ ဒါမွမဟုတ္ Object ထဲက valueေတြကိုေပါင္းေပးဖို႔ အတြက္သံုးပါတယ္ေနာက္ဆံုးက memo parameter ကေတာ့ အစဦးဆံုး(Initial State) မွာေပါင္းထည့္ေပးမယ့္ value ပါ။ဥပမာ -
_.reduce([1,2,4],function(m, el){return m+el;}, 2);//=> 9_.reduce(['der','sco', 're'],function(m, el){return m+el;}, 'un');//=> "underscore"var list = [1, 2, 3];_.reduce(list,function(a,b){return a.concat(b);},[0]);//=> [0, 1, 2, 3]
reduceRight_.reduceRight(list, iterator, memo, [context])foldr

find _.find(list, predicate, [context])detect
predicate မွာ element ေတြကိုစစ္ၿပီး boolean return ျပန္ေပးရပါတယ္predicate ႏွင့္ကိုက္ညီတဲ့ ပထမဆံုး item ကို return ျပန္ေပးပါမယ္, တစ္ခုမွမေတြ႕ရင္ undefined ကို return ျပန္ပါတယ္၊ predicate ႏွင့္ကိုက္ညီတဲ့ element ေတြ႔သည္အထိ အလုပ္လုပ္ပါတယ္ေတြ႔သြားရင္ေတာ့ ဆက္အလုပ္မလုပ္ေတာ့ပါဘူး
// Event Number_.find([1, 2, 3, 4], function(num){return num % 2 == 0;});//=> 2_.find([1, 3, 5], function(num){return num % 2 == 0;});//=> undefined
filter _.filter(list, predicate, [context])select
filter ကေတာ့ list ထဲက value အားလံုးကိုၾကည့္ပါမယ္၊predicate မွာစစ္ထားတဲ့ statement နဲ႔ကိုက္ညီတဲ့ elements အားလံုး ကို array နဲ႔ return ျပန္ေပးပါမယ္, တစ္ခုမွမေတြ႕ရင္ empty array return ျပန္ပါတယ္၊ native filter method ရိွၿပီးသားဆိုရင္ ေတာ့ ဒီ method အစား သူက အလုပ္လုပ္မွာျဖစ္ပါတယ္
// Event Numbers_.filter([1, 2, 3, 4], function(num){return num % 2 == 0;});//=> [2, 4]_.filter([1, 3, 5, 7], function(num){return num % 2 == 0;});//=> []
where _.where(list, properties)

_.where(listOfPlays, {author: "Shakespeare", year: 1611});//=> [{title: "Cymbeline", author: "Shakespeare", year: 1611},// {title: "The Tempest", author: "Shakespeare", year: 1611}]
findWhere _.findWhere(list, properties)

var persons = [{name:'koko',age:17},{name:'momo',age:20},{name:'lola',age:37}];_.findWhere(persons, {age: 17})//=> Object {name: "koko", age: 17}

// Odd Numbers_.reject([1, 2, 3, 4, 5], function(num){return num % 2 == 0;});//=> [1, 3, 5]
every _.every(list, predicate, [context])all
list ထဲက element အားလံုးကို predicate နဲ႔စစ္ပါတယ္အားလံုးကိုက္ညီမူွရိွရင္ every method က true ကို return ျပန္ပါတယ္predicate မွာစစ္လို႕ element တစ္ခုခုက မကုိက္ညီရင္ false return ျပန္ပါမယ္every native method ရိွရင္ native method ကအလုပ္လုပ္ပါလိမ့္မယ္

groupBy _.groupBy(list, iterator, [context]);
list ထဲက element ေတြကို group ဖြဲ႔ၿပီး object အျဖစ္ return ျပန္ေပးပါတယ္ဘယ္လို group ဖြဲ႔မလဲဆိုတာ iterator မွာသတ္မွတ္ေပးရပါမယ္
var list = ['man', 'pop', 'mac', 'poem', 'zip'];_.groupBy(list , function(el){return el.slice(0,2)});//=> {"ma":["man","mac"],"po":["pop","poem"],"zi":["zip"]}_.groupBy([1.3, 2.1, 2.4], function(num){return Math.floor(num);});//=> {1: [1.3], 2: [2.1, 2.4]}_.groupBy(['one', 'two', 'three'], function(el){return el.length;});//=> {3: ["one", "two"], 5: ["three"]}
